programa1 {

   f64 a, b, c, d,e ,f;
   
   b =: 2.0;
   out('sarasa');

   outer: do {
   		a =: b;
    	c =: d;
    	
    	do {						
    		e =: 1.0;
    		f =: 1.0;
    		
    		do {
    			e =: 2.0;
    			f =: 2.0;
    			
    			break :outer ;
    		} until(e > f);
    			
    	} until(a > b);
    	
    	d =: 1.0;
    	
    	continue;	
    	
   } until(a - b > c + 1);

   c =: 3.0;
}
